int: n;
array[1..3,1..2] of int: pencil;

constraint
  ((cost div pencil[1, 2]) * pencil[1, 1] >= n)
  \/
  ((cost div pencil[2, 2]) * pencil[2, 1] >= n)
  \/
  ((cost div pencil[3, 2]) * pencil[3, 1] >= n);
